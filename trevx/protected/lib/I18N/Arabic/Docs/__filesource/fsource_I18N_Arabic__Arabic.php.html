  266.     public function __call($methodName, $arguments)
  267.     {
  268.         if ($this->_compatibleMode 
  269.             && array_key_exists($methodName, $this->_compatible)
  270.         ) {

  273.         }
  274.  
  275.         // Create an instance of the ReflectionMethod class
  276.         $method = new ReflectionMethod($this->myClass, $methodName);
  277.         
  278.         $params     = array();
  279.         $parameters = $method->getParameters();
  280.  
  281.         foreach ($parameters as $parameter) {
  282.             $name  = $parameter->getName();
  283.             $value = array_shift($arguments);
  284.             
  285.             if (is_null($value) && $parameter->isDefaultValueAvailable()) {
  286.                 $value = $parameter->getDefaultValue();
  287.             }
  288.             
  289.             $params[$name] = $this->coreConvert(
  290.                 $value, 
  291.                 $this->getInputCharset(), 
  292.                 'utf-8'
  293.             );
  294.         }
  295.  
  296.         $value = call_user_func_array(array(&$this->myObject, $methodName), $params);
  297.  
  298.         if ($methodName == 'tagText') {
  299.             foreach ($value as $key=>$text) {
  300.                 $value[$key][0] = $this->coreConvert(
  301.                     $text[0], 'utf-8', $this->getOutputCharset()
  302.                 );
  303.             }
  304.         } else {
  305.             $value = $this->coreConvert($value, 'utf-8', $this->getOutputCharset());
  306.         }
  307.  
  308.         return $value;
  309.     }

  409.             
  410.             if ($outputCharset == 'windows-1256') {

  412.             }
  413.             
  414.             $convStr = iconv($inputCharset, "$outputCharset", $str);
  415.  
  416.             if ($convStr == '' && $str != '') {
  417.                 include self::getClassFile('CharsetC');
  418.  
  419.                 $c = I18N_Arabic_CharsetC::singleton();
  420.                 
  421.                 if ($inputCharset == 'cp1256') {
  422.                     $convStr = $c->win2utf($str);
  423.                 } else {
  424.                     $convStr = $c->utf2win($str);
  425.                 }
  426.             }
  427.         } else {
  428.             $convStr = $str;
  429.         }
  430.         
  431.         return $convStr;
  432.     }
